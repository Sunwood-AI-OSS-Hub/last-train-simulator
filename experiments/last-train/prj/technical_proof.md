  const response = await openai.chat.completions.create({
    model: "gpt-4",
    messages: [{
      role: "system",
      content: "勤怠改善のアドバイスをしてください"
    }, {
      role: "user",
      content: JSON.stringify(context)
    }]
  });
  return response.choices[0].message.content;
}
